\newcommand*\dgl@affil@ifdup@aux{%
  \advance\@tempcnta\@ne\relax
  \expandafter\expandafter\expandafter\def\expandafter\expandafter
    \expandafter\@tempb\expandafter\expandafter\expandafter
    {\csname @address@\@roman\@tempcnta\endcsname}%
  \ifx\@tempa\@tempb
    \expandafter\@tempswatrue

      \expandafter\expandafter\expandafter\dgl@affil@ifdup@aux
    \fi
  \fi
}
\newcommand*\dgl@affil@swap[1]{%
  \begingroup
    \def\@tempa{#1}%
    \@tempcnta\z@\relax
    \@tempcntb\z@\relax
    \dgl@affil@ifdup@aux
    \advance\dgl@affil@cnt\@ne\relax
    \dgl@affil@swap@aux
  \endgroup
}
\newcommand*\dgl@affil@swap@aux{%
  \advance\@tempcntb\@ne\relax
  \expandafter\ifnum\csname @author@affil@\@roman\@tempcntb\endcsname
    = \dgl@affil@cnt\relax
    \expandafter\xdef\csname @author@affil@\@roman\@tempcntb\endcsname{%
      \the\@tempcnta
    }%
  \fi
  \ifnum\@tempcntb<\dgl@author@cnt\relax
    \expandafter\dgl@affil@swap@aux
  \fi
}
\newcommand*\alsoaffiliation[2][\relax]{%
  \ifnum\dgl@author@cnt>\z@\relax
    \dgl@affil@ifdup{#2}{%
      \dgl@alsoaffil@find{#2}%
    }{%
      \global\advance\dgl@affil@cnt\@ne\relax
      \@tempcnta\dgl@affil@cnt\relax
      \expandafter\gdef\csname @address@\@roman\dgl@affil@cnt\endcsname
        {#2}%
      \ifx\relax#1\relax
        \expandafter\gdef\csname @affil@\@roman\dgl@affil@cnt\endcsname
          {#2}%
      \else
        \expandafter\gdef\csname @affil@\@roman\dgl@affil@cnt\endcsname
          {#1}%
      \fi
    }%
    \@ifundefined{@author@alsoaffil@\@roman\dgl@author@cnt}{%
      \expandafter\xdef\csname @author@alsoaffil@\@roman\dgl@author@cnt
        \endcsname{\the\@tempcnta}%
    }{%
      \expandafter\xdef\csname @author@alsoaffil@\@roman\dgl@author@cnt
        \endcsname{%
        \csname @author@alsoaffil@\@roman\dgl@author@cnt\endcsname
        ,\the\@tempcnta
      }%
    }%
  \else
    \dgl@warning{Affiliation with no author}%
  \fi
}
